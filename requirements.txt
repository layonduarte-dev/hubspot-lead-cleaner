cargo_mapper.py
===============

Mapeia um cargo escrito livremente para um dos valores oficiais aceitos pela
base (lista da Sólides, usada em campos de RH). Isso alimenta a propriedade
``Cargo RH``, enquanto ``Cargo`` mantém o texto original limpo.

Lógica em 4 camadas:
    1. Correspondência exata (normalizada).
    2. Dicionário de sinônimos conhecidos.
    3. Busca por palavras-chave (nível hierárquico + área RH/DP).
    4. Fallback -> "Outros".
"""

# ---------------------------------------------------------------------------
# Camada 3 — palavras-chave (nível hierárquico)
# Cada tupla: (termos, resultado_se_RH, resultado_se_DP)
# ---------------------------------------------------------------------------
_NIVEIS = [
    (["analista", "especialista"], "Analista de RH", "Analista DP"),
    (["assistente", "auxiliar"], "Assistente de RH", "Assistente de RH"),
    (["coordenador", "coordenadora", "coord"], "Coordenador de RH", "Coordenador de DP"),

    (["consultor", "consultora"], "Consultor de RH", "Consultor de DP"),
    (["estagiario", "estagiaria", "trainee"], "Estagiário de RH", "Estagiário de DP"),
]

# Termos que mapeiam direto, sem depender de área RH/DP.
_DIRETOS = [
    (["ceo", "socio", "proprietario", "empresario", "fundador",
      "diretor executivo", "presidente", "dono"], "Sócio/CEO"),
    (["recolocacao"], "Buscando recolocação"),
    (["contador", "contabil", "contabilidade"], "Contador"),
    (["professor", "docente"], "Professor"),
    (["estudante", "universitario", "graduando"], "Estudante"),
    (["coach"], "Coach"),
    (["business partner", "hrbp"], "Business Partner de RH"),
    (["recepcionista", "secretaria", "administrativo", "financeiro"],
     "Administrativo/Financeiro"),
]


def _tem_dp(norm: str) -> bool:
    return bool(re.search(r"\bdp\b", norm)) or "departamento pessoal" in norm


def _tem_rh(norm: str) -> bool:
    return bool(re.search(r"\brh\b", norm)) or "recursos humanos" in norm


def _busca_palavras_chave(norm: str) -> Optional[str]:
    for termos, resultado in _DIRETOS:
        if any(t in norm for t in termos):
            return resultado
    dp = _tem_dp(norm)
    rh = _tem_rh(norm)
    if not (dp or rh):
        return None
    for termos, res_rh, res_dp in _NIVEIS:
        if any(t in norm for t in termos):
            return res_dp if dp and res_dp else res_rh
    return None


def mapear_cargo(texto) -> str:
    """
    Mapeia um cargo livre para um dos 23 valores oficiais.
    Devolve "Outros" quando não há correspondência segura.
    """
    norm = normalizar(texto)
    if not norm:
        return "Outros"

    # 1. Correspondência exata.
    for cargo in CARGOS_OFICIAIS:
        if normalizar(cargo) == norm:
            return cargo

    # 2. Dicionário de sinônimos.
    if norm in MAPA:
        return MAPA[norm]

    # 3. Palavras-chave.
    resultado = _busca_palavras_chave(norm)
    if resultado:
        return resultado

    # 4. Fallback.
    return "Outros"
